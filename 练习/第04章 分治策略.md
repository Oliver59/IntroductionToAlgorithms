### 4.1-3

在我的计算机上当 `n0=120` 时左右为性能交叉点，从此之后递归算法将超越暴力算法 。

修改后的递归算法会将会在n0<120时性能更好，但是大于120时可能会增加性能消耗，不如传统的递归算法。

以下为C#代码：

```
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Text;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {


            List<int> aList = new List<int>();
            Random rd = new Random();
            for (int i = 0; i < 1000; i++)
            {
                aList.Add(rd.Next(5000) - 2500);
            }



            for (int i = 1; i < aList.Count; i++)
            {
                int[] A = aList.Take(i).ToArray();


                Stopwatch watch = new Stopwatch();
                watch.Start();
                var result1 = Find_Maximum_Subarray(A, 0, A.Length - 1);
                watch.Stop();
                var time1 = watch.Elapsed.TotalMilliseconds;

                Stopwatch watch2 = new Stopwatch();
                watch2.Start();
                var result2 = Find_Maximum(A, 0, A.Length - 1);
                watch2.Stop();
                var time2 = watch2.Elapsed.TotalMilliseconds;
                var faster = time1 < time2 ? "分治法" : "暴力法";
                Console.WriteLine($"数组长度：{A.Length} 更快：{faster} 分治法耗时：{time1.ToString()}ms 暴力法耗时：{time2.ToString()}ms 结果1：{result1.ToString()} 结果2：{result2.ToString()}");
            }

            Console.ReadKey();

        }


        static Result Find_Maximum_Subarray(int[] A, int low, int high)
        {
            if (low == high)
            {
                return new Result() { Left = low, Right = high, Sum = A[low] };
            }

            int mid = (low + high) / 2;

            Result leftResult = Find_Maximum_Subarray(A, low, mid);
            Result rightResult = Find_Maximum_Subarray(A, mid + 1, high);
            Result allResult = Find_Maximum_Crossing_Subarray(A, low, mid, high);

            if (leftResult.Sum >= rightResult.Sum && leftResult.Sum >= allResult.Sum)
            {
                return leftResult;
            }
            else if (rightResult.Sum >= leftResult.Sum && rightResult.Sum >= allResult.Sum)
            {
                return rightResult;
            }
            else
            {
                return allResult;
            }
        }

        static Result Find_Maximum_Crossing_Subarray(int[] A, int low, int mid, int high)
        {
            var left_sum = Int32.MinValue;

            var max_left = 0;

            var sum = 0;
            for (int i = mid; i >= low; i--)
            {
                sum += A[i];
                if (sum > left_sum)
                {
                    left_sum = sum;
                    max_left = i;
                }
            }

            var right_sum = Int32.MinValue;
            var max_right = 0;
            sum = 0;
            for (int i = mid + 1; i <= high; i++)
            {
                sum += A[i];
                if (sum > right_sum)
                {
                    right_sum = sum;
                    max_right = i;
                }
            }
            return new Result() { Left = max_left, Right = max_right, Sum = left_sum + right_sum };
        }

        /// <summary>
        /// 暴力查找
        /// </summary>
        /// <param name="A"></param>
        /// <param name="low"></param>
        /// <param name="high"></param>
        /// <returns></returns>
        static Result Find_Maximum(int[] A, int low, int high)
        {
            int sum = Int32.MinValue;
            int left = 0;
            int right = 0;
            for (int i = low; i <= high; i++)
            {
                int currentSum = 0;
                for (int j = i; j <= high; j++)
                {
                    currentSum += A[j];
                    if (sum < currentSum)
                    {
                        sum = currentSum;
                        left = i;
                        right = j;
                    }
                }
            }
            return new Result() { Left = left, Right = right, Sum = sum };
        }
    }

    class Result
    {
        public int Left { get; set; }
        public int Right { get; set; }
        public int Sum { get; set; }

        public override string ToString()
        {
            return string.Format($"Left:{Left} Right:{Right} Sum:{Sum}");
        }
    }
}
```

```csharp
/// <summary>
/// 分治算法
/// </summary>
/// <param name="A"></param>
/// <param name="low"></param>
/// <param name="high"></param>
/// <returns></returns>
static Result Find_Maximum_Subarray(int[] A, int low, int high)
{
    if (low == high)
    {
        if(A[low]<0){
			return new Result() { Left = low, Right = high, Sum = 0 };              
        }
        return new Result() { Left = low, Right = high, Sum = A[low] };
    }

    int mid = (low + high) / 2;

    Result leftResult = Find_Maximum_Subarray(A, low, mid);
    Result rightResult = Find_Maximum_Subarray(A, mid + 1, high);
    Result allResult = Find_Maximum_Crossing_Subarray(A, low, mid, high);

    if (leftResult.Sum >= rightResult.Sum && leftResult.Sum >= allResult.Sum)
    {
        return leftResult;
    }
```
